[docs]    def tearDown(self):
        time.sleep(10)
        print "Stop"
        self.client = None
        try:
            self.app.extensions['janitoo'].stop_listener()
        except RuntimeError:
            pass
        del self.app.extensions['janitoo']
        self.app = None
        if getattr(self, '_ctx', None) is not None:
            self._ctx.pop()
            del self._ctx
        JNTTBase.tearDown(self)

[docs]    def get_routes(self):
        res = {}
        for rule in self.app.url_map.iter_rules():
            res["{}".format(rule.endpoint)] = {'methods':rule.methods, 'rule':"{}".format(rule)}
        return res



[docs]    def list_routes(self):
        output = []
        routes = self.get_routes()
        for route in routes:
            line = urllib.unquote("{:50s} {:30s} {}".format(route, routes[route]['methods'], routes[route]['rule']))
            output.append(line)
        for line in sorted(output):
            print(line)
